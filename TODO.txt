function thread_task( input )
{
	output = 0;
	for( i=0;i<#input;i++ )
		output += input[i];
		
	return 2 * output;
}

results = async( thread_task, [
	[ 1, 2, 3, 4, 5 ],
	[ 3, 4, 2, 5, 1, 51, 6 ],
	[ 5, 1, 5, 2, 3, 4, 5, 32 ] ] );

outln( results[1] ); // outputs '30'
